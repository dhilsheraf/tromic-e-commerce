async function updateProductStatus(orderId, productId, status) {
    try {
        const response = await fetch(`/admin/orders/${orderId}/products/${productId}/status`, {
            method: 'PUT',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({ status })
        });

        const data = await response.json();
        if (data.success) {
            alert('Product status updated successfully');
            // Optionally, refresh the page or update the UI with the new status
        } else {
            alert(`Failed to update status: ${data.message}`);
        }
    } catch (error) {
        console.error('Error updating product status:', error);
        alert('An error occurred while updating the status');
    }
}



function updateStatus(orderId, productId, status) {
    fetch(`/admin/orders/${orderId}/products/${productId}/status`, {
        method: 'PUT',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify({ status: status })
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            alert('Product status updated successfully');
            // Optionally, you can refresh the page or update the UI dynamically
        } else {
            alert(`Failed to update status: ${data.message}`);
        }
    })
    .catch(error => {
        console.error('Error updating product status:', error);

    });
}
